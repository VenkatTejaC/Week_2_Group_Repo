Business Understanding


The AllLife bank core strategy focuses on building long-term customer relationships and increasing profitability through interest income.

Loans generate higher interest income, expanding the personal loan portfolio is critical for sustainable growth.

To improve loan conversions, the bank aims to shift from basic demographic targeting to a data-driven strategy based on customer spending behavior.

import numpy as np
from sklearn.metrics import mean_squared_error, r2_score

def evaluate_model(name, model, X_tr, X_val, y_tr, y_val):
    # Train the model
    model.fit(X_tr, y_tr)

    # Predictions
    train_preds = model.predict(X_tr)
    val_preds = model.predict(X_val)

    # Training metrics
    train_rmse = np.sqrt(mean_squared_error(y_tr, train_preds))
    train_r2 = r2_score(y_tr, train_preds)

    # Validation metrics
    val_rmse = np.sqrt(mean_squared_error(y_val, val_preds))
    val_r2 = r2_score(y_val, val_preds)

    # Print results
    print(f"{name}")
    print("Train RMSE:", train_rmse)
    print("Train R2:", train_r2)
    print("Validation RMSE:", val_rmse)
    print("Validation R2:", val_r2)
    print("-" * 30)

    return model, train_rmse, train_r2, val_rmse, val_r2
